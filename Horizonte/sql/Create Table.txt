CREATE TABLE DOKUMENT(
  DOCID BIGINT(20) PRIMARY KEY NOT NULL,
  SONGID BIGINT(20) NOT NULL,
  DATEN LONGBLOB );

CREATE TABLE MUSIKER(
  MUSIKERID BIGINT(20) PRIMARY KEY NOT NULL,
  NAME VARCHAR(128) NOT NULL,
  VORNAME VARCHAR(128) NOT NULL,
  STRASSE VARCHAR(512) NOT NULL,
  PLZ VARCHAR(32) NOT NULL,
  ORT VARCHAR(128) NOT NULL,
  EMAIL VARCHAR(256),
  TEL VARCHAR(64),
  USERNAME VARCHAR(32) NOT NULL,
  PASSWORD VARCHAR(32) NOT NULL );

CREATE TABLE COMMENT(
  COMMENTID BIGINT(20) PRIMARY KEY NOT NULL,
  DATUM VARCHAR(64) NOT NULL,
  USERNAME VARCHAR(32) NOT NULL,
  SONGID BIGINT(20) NOT NULL,
  AUFTRAGID BIGINT(20) NOT NULL,
  TEXTSTR VARCHAR(32) NOT NULL );

CREATE TABLE PROGRAMPROBE(
  PROGPROBEID BIGINT(20) PRIMARY KEY NOT NULL,
  SONGID BIGINT(20) NOT NULL,
  PROBEID BIGINT(20) NOT NULL );

CREATE TABLE PROBE(
  PROBEID BIGINT(20) PRIMARY KEY NOT NULL,
  DATUM VARCHAR(64) NOT NULL,
  ORT VARCHAR(128) NOT NULL,
  DESCRIPTION VARCHAR(512) );

CREATE TABLE PROGRAM(
  PROGID BIGINT(20) PRIMARY KEY NOT NULL,
  AUFTRID BIGINT(20) NOT NULL,
  SONGID BIGINT(20) NOT NULL );
  
CREATE TABLE AUFTRITT(
  AUFTRID BIGINT(20) PRIMARY KEY NOT NULL,
  DATUM VARCHAR(64) NOT NULL,
  ORT VARCHAR(128) NOT NULL,
  DESCRIPTION VARCHAR(512) NOT NULL,
  NOTES VARCHAR(512) NOT NULL,
  FILENAME VARCHAR(512) );

CREATE TABLE SONG(
  SONGID BIGINT(20) PRIMARY KEY NOT NULL,
  TITLE VARCHAR(128) NOT NULL,
  ORDERNMB INT(20) NOT NULL,
  NOTES VARCHAR(512) NOT NULL,
  DESCRIPTION VARCHAR(512) NOT NULL,
  FILENAME VARCHAR(512),
  CATEGORY VARCHAR(512),
  SUBCATEGORY VARCHAR(512) );
